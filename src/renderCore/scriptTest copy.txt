PUSH ANIMATION {
        relatedTo:"engineCamera",
        to:{origin:{x:.5,y:.5}},
        duration:1000,
        ease:"linear",
        id:"aCamera"}       
PUSH ANIMATION {
  relatedTo:"textBox",
  id:"a1",
  ease:"outBounce",
  duration:2000,
  to:{scale:0.8,top:0.1,left:0.1,opacity:1},
  loopback:false,
  reverse:false,
  loops:1
}
//*Triggers
//*Mover camara
PUSH TRIGGER {
      id:"t3",
      relatedTo:"moveCamera",
      onRelease:(engine)=>{
        engine.anims.get("aCamera").enabled=true;
      }
    }

      id:"t1",
      relatedTo:"button",
      onRelease:(engine)=>{
